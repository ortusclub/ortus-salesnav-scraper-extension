function doPost(e) {
  try {
    var p = JSON.parse(e.postData.contents);
    switch (p.action) {
      case "writeProfiles": return handleWrite(p);
      case "readTabs": return handleReadTabs(p);
      case "readColumns": return handleReadColumns(p);
      case "readJobs": return handleReadJobs(p);
      case "updateJob": return handleUpdateJob(p);
      case "writeBackLink": return handleWriteBackLink(p);
      case "checkSharing": return handleCheckSharing(p);
      case "getKnownProfiles": return handleGetKnownProfiles(p);
      case "writeLog": return handleWriteLog(p);
      default: return jr({success:false, error:"Unknown: "+p.action});
    }
  } catch(err) { return jr({success:false, error:err.message}); }
}

/* Verify a sheet is shared as "Anyone with the link · Editor" specifically.
 * Requires DriveApp scope — first call after redeploy may prompt re-authorization. */
function handleCheckSharing(p) {
  if (!p.sheetUrl) return jr({success:false, error:"Missing sheetUrl"});
  try {
    var m = String(p.sheetUrl).match(/\/spreadsheets\/d\/([a-zA-Z0-9_-]+)/);
    if (!m) return jr({success:false, error:"Not a valid Sheets URL"});
    var file = DriveApp.getFileById(m[1]);
    var access = file.getSharingAccess();
    var permission = file.getSharingPermission();
    var anyoneLink = (access === DriveApp.Access.ANYONE_WITH_LINK || access === DriveApp.Access.ANYONE);
    var canEdit = (permission === DriveApp.Permission.EDIT);
    var ok = anyoneLink && canEdit;
    return jr({
      success: true,
      shared: ok,
      access: String(access),
      permission: String(permission),
      anyoneLink: anyoneLink,
      canEdit: canEdit,
      reason: ok ? "OK" : (
        !anyoneLink ? "Sharing access is not 'Anyone with the link' (got '" + access + "')"
                    : "Sharing permission is not 'Editor' (got '" + permission + "')"
      )
    });
  } catch (err) {
    return jr({success:false, error:"checkSharing failed: " + err.message});
  }
}

function handleWrite(p) {
  if (!p.sheetUrl) return jr({success:false,error:"Missing sheetUrl"});
  if (!p.profiles||!p.profiles.length) return jr({success:false,error:"No profiles"});
  var ss=SpreadsheetApp.openByUrl(p.sheetUrl);
  var tab=p.tabName||"Sheet1";
  var sh=ss.getSheetByName(tab); if(!sh) sh=ss.insertSheet(tab);
  if(sh.getLastRow()===0){
    var h=["Record ID","First Name","Last Name","Domain","Priority (Company)","Priority (Role)","Priority","Lead Status","Company Name","Job Title","Linkedin Bio","First Phone","First Phone Validation","Phone Number","Whatsapp Link","Mobile Phone Number","Email","Email Verification","LinkedIn Membership ID","Location","Notes","Secondary Emails","Hubspot URL","Ortus Membership","Current Tag","Open Profile","Premium Account","Linkedin First Connections","Client Lead Status","Apollo Contact ID","Additional email addresses"];
    sh.appendRow(h);sh.getRange(1,1,1,h.length).setFontWeight("bold");sh.setFrozenRows(1);
  }
  var rows=p.profiles.map(function(r){return["",r.firstName||"",r.lastName||"","","","","","",r.company||"",r.jobTitle||"",r.publicUrl||r.profileUrl||"","","","","","",r.linkedinEmail||"","",(r.membershipId||"").toString(),r.location||"","","","","","",r.openProfile||"No",r.premium||"Unknown","","","",""]});
  var start=sh.getLastRow()+1;
  sh.getRange(start,1,rows.length,rows[0].length).setValues(rows);
  return jr({success:true,rowsWritten:rows.length});
}

/* List all VISIBLE tabs in a spreadsheet */
function handleReadTabs(p) {
  if (!p.sheetUrl) return jr({success:false,error:"Missing sheetUrl"});
  var ss = SpreadsheetApp.openByUrl(p.sheetUrl);
  var sheets = ss.getSheets();
  var tabs = [];
  for (var i = 0; i < sheets.length; i++) {
    if (!sheets[i].isSheetHidden()) {
      tabs.push({name: sheets[i].getName(), rows: sheets[i].getLastRow(), cols: sheets[i].getLastColumn()});
    }
  }
  return jr({success:true, tabs:tabs, title:ss.getName()});
}

/* Read headers + data from a specific tab */
function handleReadColumns(p) {
  if (!p.sheetUrl) return jr({success:false,error:"Missing sheetUrl"});
  var ss = SpreadsheetApp.openByUrl(p.sheetUrl);
  var sh;
  if (p.tabName && p.tabName.trim()) {
    sh = ss.getSheetByName(p.tabName.trim());
    if (!sh) return jr({success:false,error:"Tab '"+p.tabName+"' not found"});
  } else {
    sh = ss.getSheets()[0];
  }
  var data = sh.getDataRange().getValues();
  if (data.length < 1) return jr({success:true,headers:[],sampleRows:[],allRows:[],totalRows:0});
  var headers = data[0].map(function(h){return (h||"").toString();});
  var sampleRows = [];
  for (var i=1; i<Math.min(data.length,6); i++) {
    sampleRows.push(data[i].map(function(c){return (c||"").toString();}));
  }
  var allRows = [];
  for (var j=1; j<data.length; j++) {
    allRows.push(data[j].map(function(c){return (c||"").toString();}));
  }
  return jr({success:true, headers:headers, sampleRows:sampleRows, allRows:allRows, totalRows:data.length-1});
}

/* Write a result sheet link back into a column of the source sheet */
function handleWriteBackLink(p) {
  if (!p.sheetUrl || !p.tabName) return jr({success:false,error:"Missing params"});
  if (!p.row || !p.col) return jr({success:false,error:"Missing row/col"});
  var ss = SpreadsheetApp.openByUrl(p.sheetUrl);
  var sh = ss.getSheetByName(p.tabName);
  if (!sh) return jr({success:false,error:"Tab not found"});
  sh.getRange(p.row, p.col).setValue(p.value || "");
  return jr({success:true});
}

function handleReadJobs(p) {
  if(!p.jobSheetUrl) return jr({success:false,error:"Missing jobSheetUrl"});
  var ss=SpreadsheetApp.openByUrl(p.jobSheetUrl);
  var sh=ss.getSheets()[0]; var data=sh.getDataRange().getValues();
  if(data.length<2) return jr({success:true,jobs:[]});
  var jobs=[];
  for(var i=1;i<data.length;i++){
    var url=(data[i][0]||"").toString().trim(); if(!url)continue;
    jobs.push({row:i+1, salesNavUrl:url, resultSheetUrl:(data[i][1]||"").toString().trim(),
      tabName:(data[i][2]||"Sheet1").toString().trim(), status:(data[i][3]||"Pending").toString().trim(),
      profiles:parseInt(data[i][4])||0, lastRun:(data[i][5]||"").toString().trim()});
  }
  return jr({success:true,jobs:jobs});
}

function handleUpdateJob(p) {
  if(!p.jobSheetUrl||!p.row) return jr({success:false,error:"Missing params"});
  var ss=SpreadsheetApp.openByUrl(p.jobSheetUrl);
  var sh=ss.getSheets()[0];
  if(p.status) sh.getRange(p.row,4).setValue(p.status);
  if(p.profileCount!==undefined) sh.getRange(p.row,5).setValue(p.profileCount);
  sh.getRange(p.row,6).setValue(new Date().toISOString());
  if(p.errorMsg) sh.getRange(p.row,7).setValue(p.errorMsg);
  return jr({success:true,row:p.row});
}

function jr(o){return ContentService.createTextOutput(JSON.stringify(o)).setMimeType(ContentService.MimeType.JSON)}

/* Read the destination tab's LinkedIn Membership ID + Linkedin Bio columns
 * and return the raw values. Normalization happens client-side so we can
 * iterate on rules without redeploying.
 *
 * Returns: {success, ids:[...], tokens:[...], hasIdCol, hasUrlCol, error?}
 */
function handleGetKnownProfiles(p) {
  if (!p.sheetUrl) return jr({success:false, error:"Missing sheetUrl"});
  if (!p.tabName) return jr({success:false, error:"Missing tabName"});
  try {
    var ss = SpreadsheetApp.openByUrl(p.sheetUrl);
    var sheet = ss.getSheetByName(p.tabName);
    if (!sheet) return jr({success:true, ids:[], tokens:[], hasIdCol:false, hasUrlCol:false});
    var lastRow = sheet.getLastRow();
    var lastCol = sheet.getLastColumn();
    if (lastRow < 2 || lastCol < 1) return jr({success:true, ids:[], tokens:[], hasIdCol:false, hasUrlCol:false});

    var headers = sheet.getRange(1, 1, 1, lastCol).getValues()[0];
    var idColIdx = -1, urlColIdx = -1;
    var ID_HEADERS = ["linkedin membership id", "membership id", "linkedin id"];
    var URL_HEADERS = ["linkedin bio", "linkedin url", "profile url", "sales nav url"];
    for (var i = 0; i < headers.length; i++) {
      var h = String(headers[i] || "").trim().toLowerCase();
      if (idColIdx === -1 && ID_HEADERS.indexOf(h) !== -1) idColIdx = i;
      if (urlColIdx === -1 && URL_HEADERS.indexOf(h) !== -1) urlColIdx = i;
    }

    var ids = [], tokens = [];
    if (idColIdx !== -1) {
      var idVals = sheet.getRange(2, idColIdx + 1, lastRow - 1, 1).getValues();
      for (var j = 0; j < idVals.length; j++) {
        var v = String(idVals[j][0] || "").trim();
        if (v) ids.push(v);
      }
    }
    if (urlColIdx !== -1) {
      var urlVals = sheet.getRange(2, urlColIdx + 1, lastRow - 1, 1).getValues();
      for (var k = 0; k < urlVals.length; k++) {
        var u = String(urlVals[k][0] || "").trim();
        if (u) tokens.push(u);
      }
    }
    return jr({success:true, ids:ids, tokens:tokens, hasIdCol:idColIdx !== -1, hasUrlCol:urlColIdx !== -1});
  } catch (e) {
    return jr({success:false, error:String(e.message || e)});
  }
}

/* ──────────────────────────────────────────────────────────────────────
   LOGGING — per-attempt rows in __logs tab on Antonio's internal sheet.
   Hardcoded destination: scraper always writes diagnostics here, no
   matter which destination sheet the actual leads land in.
   ────────────────────────────────────────────────────────────────────── */
var LOG_SHEET_URL = "https://docs.google.com/spreadsheets/d/1MgWFXNVAH1-Xps7UgmrqFx4EYicxpzt9QfXQMfCRB8o/edit";
var LOG_TAB_NAME = "__logs";
var LOG_HEADERS = [
  "ts","mode","jobId","searchUrl","pageNum","attempt",
  "rawCount","afterDedup","skippedDupes","msExtract","tabVisible","error","scraperVer"
];

function handleWriteLog(p) {
  try {
    var ss = SpreadsheetApp.openByUrl(LOG_SHEET_URL);
    var sh = ss.getSheetByName(LOG_TAB_NAME);
    if (!sh) {
      sh = ss.insertSheet(LOG_TAB_NAME);
      sh.appendRow(LOG_HEADERS);
      sh.getRange(1, 1, 1, LOG_HEADERS.length).setFontWeight("bold");
      sh.setFrozenRows(1);
    }
    var row = [];
    for (var i = 0; i < LOG_HEADERS.length; i++) {
      var k = LOG_HEADERS[i];
      row.push(p[k] == null ? "" : String(p[k]));
    }
    sh.appendRow(row);
    return jr({success:true});
  } catch (e) {
    return jr({success:false, error:String(e.message || e)});
  }
}

/* ──────────────────────────────────────────────────────────────────────
   ONOPEN MENU — adds "Ortus Tools" menu with admin actions.
   ────────────────────────────────────────────────────────────────────── */
function onOpen() {
  SpreadsheetApp.getUi()
    .createMenu("Ortus Tools")
    .addItem("Delete all other tabs…", "deleteOtherTabs_")
    .addItem("Clear log tab", "clearLogTab_")
    .addToUi();
}

function deleteOtherTabs_() {
  var ui = SpreadsheetApp.getUi();
  var ss = SpreadsheetApp.getActive();
  var sheets = ss.getSheets();
  if (sheets.length <= 1) {
    ui.alert("Only 1 tab in this sheet — nothing to delete.");
    return;
  }
  var active = ss.getActiveSheet();
  var defaultKeep = active ? active.getName() : sheets[0].getName();
  /* Step 1: ask which tab to KEEP */
  var resp = ui.prompt(
    "Keep which tab?",
    "Type the EXACT name of the tab to KEEP. All others will be deleted.\n\n" +
    "Default (current tab): " + defaultKeep + "\n\n" +
    "Leave blank to use the default.",
    ui.ButtonSet.OK_CANCEL
  );
  if (resp.getSelectedButton() !== ui.Button.OK) return;
  var keepName = String(resp.getResponseText() || "").trim() || defaultKeep;
  var keepSheet = ss.getSheetByName(keepName);
  if (!keepSheet) {
    ui.alert('No tab named "' + keepName + '" — aborted. Check the spelling and try again.');
    return;
  }
  /* Step 2: enumerate targets, show count + sample names, confirm */
  var targets = [];
  for (var i = 0; i < sheets.length; i++) {
    if (sheets[i].getName() !== keepName) targets.push(sheets[i]);
  }
  var sample = targets.slice(0, 8).map(function(s){return s.getName();}).join(", ");
  var more = targets.length > 8 ? " (+" + (targets.length - 8) + " more)" : "";
  var confirm = ui.alert(
    "Delete " + targets.length + " tabs?",
    'This will keep "' + keepName + '" and delete:\n\n' + sample + more +
    "\n\nThis cannot be undone. Continue?",
    ui.ButtonSet.OK_CANCEL
  );
  if (confirm !== ui.Button.OK) return;
  /* Step 3: nuke */
  var deleted = 0, failed = 0, errs = [];
  for (var k = 0; k < targets.length; k++) {
    try { ss.deleteSheet(targets[k]); deleted++; }
    catch (e) { failed++; errs.push(targets[k].getName() + ": " + e.message); }
  }
  var msg = "Deleted " + deleted + " tab" + (deleted === 1 ? "" : "s") + ".";
  if (failed > 0) msg += "\n\n" + failed + " failed:\n" + errs.slice(0, 5).join("\n");
  ui.alert(msg);
}

function clearLogTab_() {
  var ui = SpreadsheetApp.getUi();
  var ss = SpreadsheetApp.getActive();
  var sh = ss.getSheetByName(LOG_TAB_NAME);
  if (!sh) { ui.alert('No "' + LOG_TAB_NAME + '" tab found.'); return; }
  var lastRow = sh.getLastRow();
  if (lastRow <= 1) { ui.alert("Log tab is already empty."); return; }
  var resp = ui.alert(
    "Clear " + (lastRow - 1) + " log rows?",
    "Keeps the header row. Cannot be undone.",
    ui.ButtonSet.OK_CANCEL
  );
  if (resp !== ui.Button.OK) return;
  sh.getRange(2, 1, lastRow - 1, sh.getLastColumn()).clearContent();
  ui.alert("Cleared " + (lastRow - 1) + " rows.");
}
